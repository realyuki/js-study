## Array.prototype
* Array.join() : 배열의 모든 요소를 연결해 하나의 문자열로 만듬.
    ```html
    <script>
    var a = ['바람', '비', '불'];
    var myVar1 = a.join();      // myVar1에 '바람,비,불'을 대입
    var myVar2 = a.join(', ');  // myVar2에 '바람, 비, 불'을 대입
    var myVar3 = a.join(' + '); // myVar3에 '바람 + 비 + 불'을 대입
    var myVar4 = a.join('');    // myVar4에 '바람비불'을 대입
    console.log(myVar1);
    </script>
    ```
